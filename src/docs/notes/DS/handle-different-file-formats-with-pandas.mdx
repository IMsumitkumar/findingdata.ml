### From Json Data

     "places_lived": ["United States", "Spain", "Germany"],
     "pet": null,

                   "pets": ["Zeus", "Zuko"]},
                  {"name": "Katie", "age": 38,
                   "pets": ["Sixes", "Stache", "Cisco"]}]
    }
    """
import json
import pandas as pd
loaded = json.loads(obj)    # dict
json_data = json.dumps(loaded)  # str
